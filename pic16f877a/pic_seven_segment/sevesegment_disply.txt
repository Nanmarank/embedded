#include <xc.h>            // Use the correct header
#define _XTAL_FREQ 10000000

#define led PORTB
#define tled TRISB 

unsigned char arr[] = {0x3f, 0x06, 0x5b, 0x4f, 0x66, 0x6d, 0x7d, 0x07, 0x7f, 0x6f};

void main()
{
    int i = 0;
    tled = 0x00;    // Set PORTB as output

    while(1)
    {
        for(i = 0; i < 10; i++)
        {
            led = arr[i];
            __delay_ms(1000);
        }
    }
}
